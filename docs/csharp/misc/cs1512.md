 <span data-ttu-id="96fef-104">在方法、属性或构造函数外部使用了 [base](../language-reference/keywords/base.md) 关键字。</span><span class="sxs-lookup"><span data-stu-id="96fef-104">The [base](../language-reference/keywords/base.md) keyword was used outside of a method, property, or constructor.</span></span>

```csharp  
// CS1512.cs  
using System;  
  
class Base {}  
  
class CMyClass : Base  
{  
    private String xx = base.ToString();   // CS1512  
    // Try putting this initialization in the constructor instead:  
    // public CMyClass()  
    // {  
    //    xx = base.ToString();  
    // }  
  
    public static void Main()  
    {  
        CMyClass z = new CMyClass();  
    }  
}  
```
