## <a name="example"></a><span data-ttu-id="43d9b-106">示例</span><span class="sxs-lookup"><span data-stu-id="43d9b-106">Example</span></span>  
 <span data-ttu-id="43d9b-107">以下示例生成 CS0643：</span><span class="sxs-lookup"><span data-stu-id="43d9b-107">The following sample generates CS0643:</span></span>

```csharp  
// CS0643.cs  
using System;  
using System.Runtime.InteropServices;  
  
[AttributeUsage(AttributeTargets.Class)]  
public class MyAttribute : Attribute  
{  
    public MyAttribute()  
    {  
    }  
  
    public int x;  
}  
  
[MyAttribute(x = 5, x = 6)]   // CS0643, error setting x twice  
// try the following line instead  
// [MyAttribute(x = 5)]  
class MyClass  
{  
}  
  
public class MainClass  
{  
    public static void Main ()  
    {  
    }  
}  
```
