 <span data-ttu-id="286e5-104">接口中的事件不能有初始值设定项。</span><span class="sxs-lookup"><span data-stu-id="286e5-104">An event in an interface cannot have an initializer.</span></span> <span data-ttu-id="286e5-105">有关详细信息，请参阅[接口](../../csharp/programming-guide/interfaces/index.md)。</span><span class="sxs-lookup"><span data-stu-id="286e5-105">For more information, see [Interfaces](../../csharp/programming-guide/interfaces/index.md).</span></span>  
  
 <span data-ttu-id="286e5-106">下面的示例生成 CS0068：</span><span class="sxs-lookup"><span data-stu-id="286e5-106">The following sample generates CS0068:</span></span>  
  
```csharp  
// CS0068.cs  
  
delegate void MyDelegate();  
  
interface I  
{  
   event MyDelegate d = new MyDelegate(M.f);   // CS0068  
   // try the following line instead  
   // event MyDelegate d2;  
}  
  
class M  
{  
   event MyDelegate d = new MyDelegate(M.f);  
  
   public static void f()  
   {  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
