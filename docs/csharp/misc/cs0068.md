 <span data-ttu-id="58b01-104">接口中的事件不能有初始值设定项。</span><span class="sxs-lookup"><span data-stu-id="58b01-104">An event in an interface cannot have an initializer.</span></span> <span data-ttu-id="58b01-105">有关详细信息，请参阅[接口](../../csharp/programming-guide/interfaces/index.md)。</span><span class="sxs-lookup"><span data-stu-id="58b01-105">For more information, see [Interfaces](../../csharp/programming-guide/interfaces/index.md).</span></span>

```csharp  
// CS0068.cs

{  
   event MyDelegate d = new MyDelegate(M.f);   // CS0068  
   // try the following line instead  
   // event MyDelegate d2;

{  
   event MyDelegate d = new MyDelegate(M.f);  
  
   public static void f()  
   {  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
